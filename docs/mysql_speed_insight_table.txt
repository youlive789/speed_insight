CREATE TABLE MASTER (
	ID INT PRIMARY KEY,
	URL VARCHAR(300),
	DATE DATETIME 
);

CREATE TABLE MAIN_SCORE (
	ID INT,
	FIRST_CONTENTFUL_PAINT FLOAT,
	SPEED_INDEX FLOAT,
	LARGEST_CONTENTFUL_PAINT FLOAT,
	TIME_TO_INTERACTIVE FLOAT,
	TOTAL_BLOCKING_TIME FLOAT,
	CUMULATIVE_LAYOUT_SHIFT FLOAT,
	FOREIGN KEY(ID)
	REFERENCES MASTER(ID) ON UPDATE CASCADE
);

CREATE TABLE SUMMARY (
	ID INT,
	NUM_REQUEST INT,
	NUM_SCRIPTS INT,
	NUM_FONTS INT,
	NUM_TASKS INT,
	RTT INT,
	THROUGHPUT FLOAT,
	MAX_RTT FLOAT,
	MAX_SERVER_LATENCY FLOAT,
	TOTAL_BYTE_WEIGHT FLOAT,
	TOTAL_TASK_TIME FLOAT,
	MAIN_DOCUMENT_TRANSFER_SIZE FLOAT,
	FOREIGN KEY(ID)
	REFERENCES MASTER(ID) ON UPDATE CASCADE
);

CREATE TABLE NETWORK_SERVER_LATENCY (
	ID INT,
	ORIGIN VARCHAR(300),
	SERVER_RESPONSE_TIME FLOAT,
	FOREIGN KEY(ID)
	REFERENCES MASTER(ID) ON UPDATE CASCADE
);

CREATE TABLE RESOURCE_SUMMARY (
	ID INT,
	RESOURCE_TYPE VARCHAR(200),
	REQUEST_COUNT INT,
	SIZE INT,
	FOREIGN KEY(ID)
	REFERENCES MASTER(ID) ON UPDATE CASCADE
);

CREATE TABLE THIRD_PARTY_SUMMARY (
	ID INT,
	TYPE VARCHAR(200),
	TEXT VARCHAR(300),
	URL VARCHAR(300),
	TRANSFER_SIZE INT,
	MAIN_THREAD_TIME FLOAT,
	BLOCKING_TIME FLOAT,
	FOREIGN KEY(ID)
	REFERENCES MASTER(ID) ON UPDATE CASCADE
);
